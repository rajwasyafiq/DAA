Step 2 Bagi nilai m dengan n, masukkan nilai sisanya ke r.
Step 3 Masukan nilai n ke m dan nilai r ke n. Kembali ke Step 1.

r <- m mod n

return m

#include <stdio.h>

int FPB(int a, int b){
    int r = 0;
    while(b!=0){
        r = a % b;
        a = b;
        b = r;
    }

    return a;
}

int main(){
int a=120;
int b=88;
int KPK=0;

    KPK= a*b / FPB(a, b);

}
